\section{A basic model private value model}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\ft{Private value auction}
\setbeamertemplate{itemize items}[triangle]

\begin{align}
E(S_i) &= (v_i - B(w_i)) F^{n-1}(w_i) + (0) (1-F^{n-1}(w_i)) \\
&\rightarrow B'(v_i) = (n-1)(v_i-B(v_i))\frac{f(v_i)}{F(v_i)} \\
B(v_i) &= v_i - \frac{\int_{\underline{v}}^{\overline{v}}F^{n-1}(t)dt }{F^{n-1}(v_i)} \\
\end{align}

If uniform

\begin{align}
B(v_i) = \frac{n-1}{n}v_i
\end{align}

}

\frame{\ft{Revenue equivalence}
\setbeamertemplate{itemize items}[triangle]

\bi
\item All direct mechanisms have the same expected revenue
\item Limitation 1: Risk averse agents shade less in Dutch and first price
\item Limitation 2: Asymmetric distributions, it may result in inefficiency 
\item Limitation 3: Similar to two part tarrif, reserve price helps
\ei

}



\frame{\ft{Common value versus private value}
\setbeamertemplate{itemize items}[triangle]

\bi
\item Common value: There is an objective value to the good
\item Private value: Value is subjective
\item Common value: Winner's curse
\item Common value also led to the no trade theorem
\item Example: Imagine bidding for a wallet
\ei

}



\frame{\ft{Linkage principle}
\setbeamertemplate{itemize items}[triangle]

\bi
\item The auction designer has an incentive to maximize information of participants
\item Reason: If bidders assymetrically informed, more opportunities for surplus
\item If bidders have to look for information themselves, they lower their bids
\ei

}

\frame{\ft{Contests}
\setbeamertemplate{itemize items}[triangle]

\bi
\item What if everyone pays? 
\item Marginally willing to bid more than your valuation?
\item $v_i F^{N-1}(b^{-1}(b_i)) - b_i$
\item Result: People bid less
\ei

}

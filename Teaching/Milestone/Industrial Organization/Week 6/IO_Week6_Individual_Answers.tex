\section*{Industrial Organization, Week 6 Individual Answers}


\subsection*{Answers to questions}
\begin{align*}
&\text{Indifferent consumer} ~
&& r-t(x-l_1)^2-p_1 = r-t(x-l_2)^2-p_2
\\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
&\text{Solving for x should give us} ~
&&  \overline{x} =  \frac{p_2-p_1+t(l_2^2-l_1^2)}{2t(l_2-l_1)}
\\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
&\text{We now set the profit function of 1} ~
&&  \pi_1= p \left( \frac{p_2-p_1+t(l_2^2-l_1^2)}{2t(l_2-l_1)} \right)
\\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
&\text{The price is then} ~
&&  \frac{\delta \pi_1}{\delta p_1}= 0 \rightarrow p_1 = \frac{t}{3}\left( l_2(2+ l_2)- l_1(2+ l_1)  \right)
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\\
&\text{By symmetry} ~
&&  \frac{\delta \pi_2}{\delta p_2}= 0 \rightarrow p_2 = \frac{t}{3}\left(  l_1(2+ l_1)- l_2(2+ l_2)  \right)
\\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
&\text{Plug this into profit} ~
&&  \pi_1 = \frac{1}{18} t ( l_2+ l_1+2) \left(l_2^2+2 l_2- l_1 ( l_1+2)\right) \\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
&\text{Take derivative} ~
&& \frac{\delta \pi_1}{\delta l_1}  = \frac{t}{18} \left(l_2^2-2l_2 l_1-3 l_1^2-8l_1-4\right) \\
\end{align*}

Which is always negative when location is contained within 0-1, which means that profit is maximized when firm 1 places itself all the way to the left! So the equilibrium is that firms maximally differentiate!
